        let currentQuiz = [];
        let currentQuestionIdx = 0;
        let score = 0;
        let answered = false;

        function startQuiz(category) {
            currentQuiz = database[category];
            currentQuestionIdx = 0;
            score = 0;
            document.getElementById('dashboard').classList.add('hidden');
            document.getElementById('quiz-container').classList.remove('hidden');
            document.getElementById('question-box').classList.remove('hidden');
            document.getElementById('results-screen').classList.add('hidden');
            updateScoreDisplay();
            showQuestion();
        }

        function showDashboard() {
            document.getElementById('dashboard').classList.remove('hidden');
            document.getElementById('quiz-container').classList.add('hidden');
        }

        function showQuestion() {
            answered = false;
            const question = currentQuiz[currentQuestionIdx];
            document.getElementById('progress-text').innerText = `Pregunta ${currentQuestionIdx + 1} de ${currentQuiz.length}`;
            document.getElementById('question-text').innerText = question.q;
            document.getElementById('feedback').classList.add('hidden');
            document.getElementById('next-btn').classList.add('hidden');
            
            const optionsDiv = document.getElementById('options-container');
            optionsDiv.innerHTML = '';
            
            question.o.forEach((opt, idx) => {
                const btn = document.createElement('button');
                btn.className = "w-full text-left p-4 border-2 border-slate-200 rounded-xl hover:bg-slate-50 transition-all font-medium";
                btn.innerText = opt;
                btn.onclick = () => checkAnswer(idx, btn);
                optionsDiv.appendChild(btn);
            });
        }

        function checkAnswer(idx, btn) {
            if (answered) return; // Evitar múltiples clics
            answered = true;

            const question = currentQuiz[currentQuestionIdx];
            const feedback = document.getElementById('feedback');
            const options = document.getElementById('options-container').children;

            feedback.classList.remove('hidden');
            
            if(idx === question.a) {
                score++;
                btn.classList.add('correct-ans');
                feedback.className = "mt-6 p-4 rounded-lg bg-emerald-50 text-emerald-700 font-semibold border border-emerald-200";
                feedback.innerHTML = `<i class="fas fa-check-circle mr-2"></i> ${question.f}`;
            } else {
                btn.classList.add('wrong-ans');
                options[question.a].classList.add('correct-ans'); // Mostrar la correcta
                feedback.className = "mt-6 p-4 rounded-lg bg-rose-50 text-rose-700 font-semibold border border-rose-200";
                feedback.innerHTML = `<i class="fas fa-times-circle mr-2"></i> ¡Incorrecto! La respuesta correcta era: ${question.o[question.a]}`;
            }
            
            updateScoreDisplay();
            document.getElementById('next-btn').classList.remove('hidden');
        }

        function updateScoreDisplay() {
            document.getElementById('score-counter').innerText = `Puntos: ${score}`;
        }

        function nextQuestion() {
            currentQuestionIdx++;
            if(currentQuestionIdx < currentQuiz.length) {
                showQuestion();
            } else {
                showResults();
            }
        }

        function showResults() {
            document.getElementById('question-box').classList.add('hidden');
            document.getElementById('results-screen').classList.remove('hidden');
            const percent = (score / currentQuiz.length) * 100;
            document.getElementById('final-score-text').innerText = `Lograste ${score} de ${currentQuiz.length} puntos (${percent}% de efectividad).`;
            document.getElementById('result-icon').innerText = percent >= 70 ? '🏆' : '📚';
        }
    </script>

let timeLeft = 30;
let timerInterval;

function startTimer() {
    clearInterval(timerInterval); // Limpiar cualquier timer previo
    timeLeft = 30;
    const timerText = document.getElementById('timer-text');
    const timerCont = document.getElementById('timer-container');
    
    updateTimerDisplay(timerText, timerCont);

    timerInterval = setInterval(() => {
        timeLeft--;
        updateTimerDisplay(timerText, timerCont);

        if (timeLeft <= 0) {
            clearInterval(timerInterval);
            handleTimeout();
        }
    }, 1000);
}

function updateTimerDisplay(textEl, contEl) {
    textEl.innerText = timeLeft;
    if (timeLeft <= 10) {
        contEl.classList.replace('border-indigo-100', 'border-rose-500');
        textEl.classList.replace('text-indigo-600', 'text-rose-600');
    } else {
        contEl.className = "relative flex items-center justify-center w-16 h-16 rounded-full border-4 border-indigo-100 transition-colors duration-300";
        textEl.className = "text-xl font-bold text-indigo-600";
    }
}

function handleTimeout() {
    if (answered) return;
    const feedback = document.getElementById('feedback');
    const question = currentQuiz[currentQuestionIdx];
    
    answered = true;
    feedback.classList.remove('hidden');
    feedback.className = "mt-6 p-4 rounded-lg bg-amber-50 text-amber-700 font-semibold border border-amber-200";
    feedback.innerHTML = `<i class="fas fa-clock mr-2"></i> ¡Se acabó el tiempo! La respuesta era: ${question.o[question.a]}`;
    
    document.getElementById('next-btn').classList.remove('hidden');
}

        let currentQuiz = [];
        let currentQuestionIdx = 0;
        let score = 0;
        let answered = false;
        let timeLeft = 30;
        let timerInterval;

        function startQuiz(category) {
            currentQuiz = database[category];
            currentQuestionIdx = 0;
            score = 0;
            document.getElementById('dashboard').classList.add('hidden');
            document.getElementById('quiz-container').classList.remove('hidden');
            document.getElementById('question-box').classList.remove('hidden');
            document.getElementById('results-screen').classList.add('hidden');
            updateScoreDisplay();
            showQuestion();
        }

        function showDashboard() {
            clearInterval(timerInterval);
            document.getElementById('dashboard').classList.remove('hidden');
            document.getElementById('quiz-container').classList.add('hidden');
        }

        function showQuestion() {
            answered = false;
            const question = currentQuiz[currentQuestionIdx];
            
            // UI
            document.getElementById('progress-text').innerText = `Pregunta ${currentQuestionIdx + 1} de ${currentQuiz.length}`;
            document.getElementById('question-text').innerText = question.q;
            document.getElementById('feedback').classList.add('hidden');
            document.getElementById('next-btn').classList.add('hidden');
            updateProgressBar();
            
            const optionsDiv = document.getElementById('options-container');
            optionsDiv.innerHTML = '';
            
            question.o.forEach((opt, idx) => {
                const btn = document.createElement('button');
                btn.className = "w-full text-left p-4 border-2 border-slate-200 rounded-xl hover:bg-slate-50 transition-all font-medium";
                btn.innerText = opt;
                btn.onclick = () => checkAnswer(idx, btn);
                optionsDiv.appendChild(btn);
            });

            startTimer();
        }

        function startTimer() {
            clearInterval(timerInterval);
            timeLeft = 30;
            updateTimerUI();
            timerInterval = setInterval(() => {
                timeLeft--;
                updateTimerUI();
                if (timeLeft <= 0) {
                    clearInterval(timerInterval);
                    handleTimeout();
                }
            }, 1000);
        }

        function updateTimerUI() {
            const textEl = document.getElementById('timer-text');
            const contEl = document.getElementById('timer-container');
            textEl.innerText = timeLeft;
            if (timeLeft <= 10) {
                contEl.className = "w-14 h-14 rounded-full border-4 border-rose-500 flex items-center justify-center animate-pulse";
                textEl.className = "text-lg font-bold text-rose-600";
            } else {
                contEl.className = "w-14 h-14 rounded-full border-4 border-indigo-100 flex items-center justify-center";
                textEl.className = "text-lg font-bold text-indigo-600";
            }
        }

        function checkAnswer(idx, btn) {
            if (answered) return;
            answered = true;
            clearInterval(timerInterval);

            const question = currentQuiz[currentQuestionIdx];
            const feedback = document.getElementById('feedback');
            const options = document.getElementById('options-container').children;

            feedback.classList.remove('hidden');
            
            if(idx === question.a) {
                score++;
                btn.classList.add('correct-ans');
                feedback.className = "mt-6 p-4 rounded-lg bg-emerald-50 text-emerald-700 font-semibold border border-emerald-200";
                feedback.innerHTML = `<i class="fas fa-check-circle mr-2"></i> ${question.f}`;
            } else {
                btn.classList.add('wrong-ans');
                options[question.a].classList.add('correct-ans');
                feedback.className = "mt-6 p-4 rounded-lg bg-rose-50 text-rose-700 font-semibold border border-rose-200";
                feedback.innerHTML = `<i class="fas fa-times-circle mr-2"></i> Incorrecto. La respuesta era: ${question.o[question.a]}`;
            }
            
            updateScoreDisplay();
            document.getElementById('next-btn').classList.remove('hidden');
        }

        function handleTimeout() {
            answered = true;
            const question = currentQuiz[currentQuestionIdx];
            const feedback = document.getElementById('feedback');
            const options = document.getElementById('options-container').children;
            
            options[question.a].classList.add('correct-ans');
            feedback.classList.remove('hidden');
            feedback.className = "mt-6 p-4 rounded-lg bg-amber-50 text-amber-700 font-semibold border border-amber-200";
            feedback.innerHTML = `<i class="fas fa-clock mr-2"></i> ¡Tiempo agotado! La respuesta era: ${question.o[question.a]}`;
            document.getElementById('next-btn').classList.remove('hidden');
        }

        function updateScoreDisplay() {
            document.getElementById('score-counter').innerText = `Puntos: ${score}`;
        }

        function updateProgressBar() {
            const progress = (currentQuestionIdx / currentQuiz.length) * 100;
            document.getElementById('progress-bar').style.width = `${progress}%`;
        }

        function nextQuestion() {
            currentQuestionIdx++;
            if(currentQuestionIdx < currentQuiz.length) {
                showQuestion();
            } else {
                showResults();
            }
        }

        function showResults() {
            document.getElementById('progress-bar').style.width = `100%`;
            document.getElementById('question-box').classList.add('hidden');
            document.getElementById('results-screen').classList.remove('hidden');
            const percent = (score / currentQuiz.length) * 100;
            document.getElementById('final-score-text').innerText = `Lograste ${score} de ${currentQuiz.length} puntos (${percent}%).`;
            document.getElementById('result-icon').innerText = percent >= 80 ? '🏆' : (percent >= 50 ? '🥈' : '📚');
        }
